<script>
const { CozeWebSDK } = window;
// A=任务组ID，B=社交组ID（保持你原本正确的bot_id不动）
const botA = "任务机器人ID";
const botB = "社交机器人ID";
const PAT_TOKEN = "你的PAT密钥";
let sdkIns = null;

// 每次先彻底销毁旧实例，再新建机器人
function initBot(botId){
    // 强制销毁之前弹窗
    if(sdkIns){
        sdkIns.destroy();
        sdkIns = null;
    }
    // 新建对应分组机器人
    sdkIns = new CozeWebSDK({
        botId: botId,
        layout: "pc",
        isOpen: true,
        auth:{type:"token",token:PAT_TOKEN},
        asstBtn:{isNeed:true}
    })
}

// 页面加载优先执行
window.onload = function(){
  const hash = location.hash;
  if(hash === '#groupA') initBot(botA);
  if(hash === '#groupB') initBot(botB);
}
</script>
